39void Finger::update(const std::vector<dm::JointState>& state)
40
41{
42 // Update state vector
43 m_state_vec = state;
44
45 // Resize states
46 m_local_transform.resize(m_state_size);
47 m_global_transform.resize(m_state_size);
48
49 /* Loop through state vector components and
50 generate transformation matrices */
51 for (size_t i = 0; i < m_state_vec.size(); i++)
52 {
53 /*********** Local transformation ***********/
54 // Initialize transformation matrices
55 Eigen::Matrix4d local_transform = Eigen::Matrix4d::Identity();
56
57 // Position vector
58 local_transform.block<3, 1>(0, 3) = m_state_vec.at(i).position;
59
60 // Rotation matrix
61 local_transform.block<3, 3>(0, 0) =
62 EulerRotations::rotation(m_state_vec.at(i).euler);
63
64 // Push back locak transform
65 m_local_transform.at(i) = local_transform;
66
67 /*********** Global transformation ***********/
68 if (i == 0) {
69 // Global transform
70 m_global_transform.at(i) = m_local_transform.at(i);
71 }
72 else {
73 // Global transform
74 m_global_transform.at(i) = m_global_transform.at(i-1) *
75 m_local_transform.at(i);
76 }
77 }
78
79 // Resize vertices data
80 m_vertices_data.resize(2 * m_state_size);
81
82 // Loop through global transformation matrices
83 for (size_t i = 0; i < m_global_transform.size(); i++)
84 {
85 // Get global transformation matrix
86 Eigen::Matrix4d t_mat = m_global_transform.at(i);
87
88 /****************** Transform vertices ********************/
89 // Joint vertices
90 Eigen::MatrixXd joint_vert =
91 (t_mat * m_vertices_data_oh.at(2*i).transpose()).transpose();
92
93 // Link vertices
94 Eigen::MatrixXd link_vert =
95 (t_mat * m_vertices_data_oh.at(2*i+1).transpose()).transpose();
96
97 // Push back vertices
98 m_vertices_data.at(2*i) = joint_vert.leftCols<3>();
99 m_vertices_data.at(2*i+1) = link_vert.leftCols<3>();

172void Finger::postprocess_meshes(void)
173{
174 // Scale meshes
175 for (size_t i = 0; i < m_vertices_data_o.size(); i++)
176 {
177 /************** Scale data *******************/
178 Eigen::Matrix3d scale_matrix;
179 scale_matrix << m_geom_scales.at(i), 0.0, 0.0,
180 0.0, m_geom_scales.at(i), 0.0, 0.0, 0.0, m_geom_scales.at(i);
181
182 m_vertices_data_o.at(i) = (scale_matrix *
183 m_vertices_data_o.at(i).transpose()).transpose();
184
185 /************** Generate homogeneous vertices data *******************/
186 // Get mesh i
187 Eigen::MatrixXd mesh_i = m_vertices_data_o.at(i);
188
189 // Initialize homogeneous coordinates
190 Eigen::MatrixXd mesh_i_oh = Eigen::MatrixXd::Ones(mesh_i.rows(),
191 mesh_i.cols() + 1);
192
193 // Generate matrix of homogeneous coordinates
194 mesh_i_oh.col(0) = mesh_i.col(0);
195 mesh_i_oh.col(1) = mesh_i.col(1);
196 mesh_i_oh.col(2) = mesh_i.col(2);
